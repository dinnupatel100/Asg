####### GIT commands  #######
git init (it is used to initialize directory)
git checkout -b <-branch name-> (used to create branch)
git add <-file name-> (used to add the changed/modified file to the staged)
git add . (used to add all the files to the staged)
git commit -m "message" (used to commit the changes with the message)
git stash (used to store the changes at local machine which can be used later)
git diff <-branch name-> (used to compare the current branches with the mentioned branch)
git status (used to check the status of the files)
git branch (used to get all the branches)
git branch -d <-branch name-> (used to delete the branch)
git checkout <-branch name-> (used to navigate the branches)
git push origin <-branch name-> (used to push the local repo to remote)
git remote add origin <-repo link-> (used to link the local repo with remote repo)
git log (used to see all the commit in particular branch)
git show <-commit id-> (used to see all the changes happens in repo)
git branch -m <-branch name-> (used to rename the current branch)
git reset --soft HEAD~1 (used to change the message of the recent commit)
git reset --hard HEAD~1 (used to discard all the changes made in the file)
